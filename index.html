package com.example.Universal.Service; import java.io.BufferedWriter; import java.io.File; import java.io.FileWriter; import java.io.IOException; import java.util.Map; import java.util.stream.Collectors; import org.springframework.stereotype.Service; import tools.jackson.databind.ObjectMapper; import tools.jackson.databind.exc.JsonNodeException; @Service public class CommonProcessingService { public void process(String channelCode, String channelIdentifier, Object validatedData) { try { // Convert object to pipe-separated TXT String txtData = convertToPipeSeparated(validatedData); // Generate file path String fileName = channelCode + "_" + System.currentTimeMillis() + "_" + java.util.UUID.randomUUID() + ".txt"; String filePath = "output/" + channelIdentifier + "/" + fileName; File file = new File(filePath); file.getParentFile().mkdirs(); // Write using BufferedWriter try (BufferedWriter writer = new BufferedWriter(new FileWriter(file))) { writer.write(txtData); writer.flush(); } // Audit logging System.out.println("TXT generated for " + channelIdentifier + " at " + filePath); } catch (IOException e) { throw new RuntimeException("Error generating TXT file", e); } } private String convertToPipeSeparated(Object dto) throws JsonNodeException { ObjectMapper mapper = new ObjectMapper(); Map map = mapper.convertValue(dto, Map.class); return map.values().stream() .map(String::valueOf) .collect(Collectors.joining("|")); } }
